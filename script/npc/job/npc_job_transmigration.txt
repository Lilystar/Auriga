// 転生二次転職ファンクション
//	callfunc "JobChanger","Name",1stClass,2ndClass
//------------------------------------------

function	script	JobChanger	{
	if(JobLevel < 40 || Class != getarg(1) || OLD_CLASS != getarg(2))
		return;
	mes "["+getarg(0)+"]";
	mes "本来の姿を取り戻し、";
	mes "世界に新たな光をもたらす";
	mes "時が訪れました。";
	mes "新たな姿であなたの道を";
	mes "歩んでください。";
	next;
	mes "["+getarg(0)+"]";
	mes getarg(0)+"に";
	mes "転職なさいますか？";
	next;
	if(select("いいえ","はい")==1) {
		mes "["+getarg(0)+"]";
		mes "準備ができましたら、";
		mes "再度私をお訪ねください。";
		mes "英雄に祝福を！";
		close;
	}
	if(SkillPoint) {
		mes "["+getarg(0)+"]";
		mes "まだ学ばなければならないことが";
		mes "おありのようですね。";
		mes "スキルポイントを全て消費してから";
		mes "お越しください。";
		close;
	}
	unequip;
	jobchange getarg(2),1;
	if(Class==18 && SKILL_BIOETHICS==13) {
		skill 238,1,0;
		set SKILL_BIOETHICS,0;
	}
	mes "["+getarg(0)+"]";
	mes "お祝い申し上げます。";
	mes "地上へ戻り、あなたのお力で";
	mes "人々をお導きください。";
	close;
}
